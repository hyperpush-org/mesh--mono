```rust
for item in tree.items() {
    if let Item::TypeAliasDef(alias_def) = &item {
        if let Some(name) = alias_def.name().and_then(|n| n.text()) {
            if alias_def.visibility().is_some() {
                if let Some(def) = typeck.type_registry.type_aliases.get(&name) {

            } else {

            }
        }
    }
}
```

**1. Create `tests/e2e/sqlite_upsert_subquery_runtime.mpl`:**

The fixture should:
1. Open an in-memory SQLite database via `Sqlite.open(":memory:")`
2. Create two tables: `organizations` and `projects` (with a UNIQUE constraint on `projects(id)`)
3. Insert seed data using raw SQL
4. **Test UPS-01 via `Repo.insert_or_update`** -- call the function directly with a Map of fields, conflict targets, and update fields. This invokes `mesh_repo_insert_or_update` which calls `build_upsert_sql_pure` to generate `INSERT INTO ... ON CONFLICT (...) DO UPDATE SET ... RETURNING *` SQL
5. **Test UPS-02 via `Repo.delete_where_returning`** -- call the function directly with a Query. This invokes `mesh_repo_delete_where_returning` which generates `DELETE FROM ... WHERE ... RETURNING *` SQL
6. Test UPS-03 subquery WHERE using raw SQL (subquery SQL is tested for semantics; the query builder wrapping was proven in Plan 01)
7. Print results for assertion

```mesh
fn main() do
  let db = Sqlite.open(":memory:")?

  # Create tables (id is PRIMARY KEY = implicit UNIQUE constraint for ON CONFLICT)
  Sqlite.execute(db, "CREATE TABLE organizations (id TEXT PRIMARY KEY, name TEXT)", [])?
  Sqlite.execute(db, "CREATE TABLE projects (id TEXT PRIMARY KEY, org_id TEXT, name TEXT, status TEXT)", [])?

  # Seed data
  Sqlite.execute(db, "INSERT INTO organizations VALUES ('org1', 'Acme Corp')", [])?
  Sqlite.execute(db, "INSERT INTO organizations VALUES ('org2', 'Beta Inc')", [])?
  Sqlite.execute(db, "INSERT INTO projects VALUES ('p1', 'org1', 'Alpha', 'active')", [])?
  Sqlite.execute(db, "INSERT INTO projects VALUES ('p2', 'org1', 'Beta', 'active')", [])?
  Sqlite.execute(db, "INSERT INTO projects VALUES ('p3', 'org2', 'Gamma', 'archived')", [])?

  # === UPS-01: Upsert via Repo.insert_or_update ===
  # This calls mesh_repo_insert_or_update which builds:
  #   INSERT INTO "projects" ("id","org_id","name","status") VALUES ($1,$2,$3,$4)
  #   ON CONFLICT ("id") DO UPDATE SET "name" = EXCLUDED."name", "status" = EXCLUDED."status"
  #   RETURNING *

  # First call: insert new row (p4 does not exist yet)
  let fields1 = Map.new()
    |> Map.put("id", "p4")
    |> Map.put("org_id", "org2")
    |> Map.put("name", "Delta")
    |> Map.put("status", "active")
  let res1 = Repo.insert_or_update(db, "projects", fields1, ["id"], ["name", "status"])
  case res1 do
    {:ok, row} -> println("upsert_insert:" <> Map.get(row, "name"))
    {:error, e} -> println("upsert_insert_err:" <> e)
  end

  # Second call: update existing row (p4 now exists, ON CONFLICT triggers DO UPDATE SET)
  let fields2 = Map.new()
    |> Map.put("id", "p4")
    |> Map.put("org_id", "org2")
    |> Map.put("name", "Delta-Updated")
    |> Map.put("status", "active")
  let res2 = Repo.insert_or_update(db, "projects", fields2, ["id"], ["name", "status"])
  case res2 do
    {:ok, row} -> println("upsert_update:" <> Map.get(row, "name"))
    {:error, e} -> println("upsert_update_err:" <> e)
  end

  # Verify total count is 4 (3 seed + 1 new, upsert did NOT create a duplicate)
  let count_rows = Sqlite.query(db, "SELECT count(*) AS cnt FROM projects", [])?
  let cr = List.head(count_rows)
  println("upsert_count:" <> Map.get(cr, "cnt"))

  # === UPS-02: Delete with RETURNING via Repo.delete_where_returning ===
  # This calls mesh_repo_delete_where_returning which builds:
  #   DELETE FROM "projects" WHERE "id" = $1 RETURNING *
  # SQLite 3.35.0+ supports RETURNING.
  let del_q = Query.from("projects") |> Query.where(:id, "p3")
  let del_res = Repo.delete_where_returning(db, "projects", del_q)
  case del_res do
    {:ok, rows} ->
      let del_count = List.length(rows)
      println("delete_ret_count:" <> Int.to_string(del_count))
      let del_row = List.head(rows)
      println("delete_ret_name:" <> Map.get(del_row, "name"))
    {:error, e} -> println("delete_ret_err:" <> e)
  end

  # Verify p3 is actually gone
  let after_rows = Sqlite.query(db, "SELECT count(*) AS cnt FROM projects WHERE id = 'p3'", [])?
  let after = List.head(after_rows)
  println("delete_after_count:" <> Map.get(after, "cnt"))

  # === UPS-03: Subquery WHERE test ===
  # Find all projects belonging to org1 using a subquery
  let sub_rows = Sqlite.query(db, "SELECT name FROM projects WHERE org_id IN (SELECT id FROM organizations WHERE name = 'Acme Corp') ORDER BY name ASC", [])?
  println("subquery_count:" <> Int.to_string(List.length(sub_rows)))
  let sr1 = List.head(sub_rows)
  println("subquery_first:" <> Map.get(sr1, "name"))

  Sqlite.close(db)
  println("done")
end
```
